233  template <typename... Args>
234  std::string Properties::get(const std::string& name, Args&&... args) const
235  {
236  auto itr = m_properties.find(name);
237 
238  if (itr != std::end(m_properties))
239  {
240  Property_t p = itr->second;
241 
242  std::stringstream ss;
243  std::vector<std::string> results;
244 
245  this->process(ss, args...);
246 
247  while (ss.good())
248  {
249  std::string sub;
250  std::getline(ss, sub, ',');
251  results.push_back(sub);
252  }
253 
254  for (auto itr = std::rbegin(p.parameters); itr != std::rend(p.parameters); ++itr)
255  {
256  std::size_t start = (*itr).first;
257  std::size_t end = results.at((*itr).second).length();
258 
259  std::string param = results.at((*itr).second);
260  p.value.insert(start, param);
261 
262  p.value.erase(start + end, 2);
263  p.value.erase(start - 1, 1);
264  }
265 
266  return p.value;
267  }
268 
269  m_log.warning(m_log.function(__FUNCTION__, name), "Property cannot be found in properties file.");
270  return std::string();
271  }
