191  std::string Properties::get(const std::string& name, Args&&... args) const
192  {
193  auto itr = m_properties.find(name);
194 
195  if (itr != std::end(m_properties))
196  {
197  Property_t p = itr->second;
198 
199  std::stringstream ss;
200  std::vector<std::string> results;
201 
202  this->process(ss, args...);
203 
204  while (ss.good())
205  {
206  std::string sub;
207  std::getline(ss, sub, ',');
208  results.push_back(sub);
209  }
210 
211  for (auto itr = std::rbegin(p.parameters); itr != std::rend(p.parameters); ++itr)
212  {
213  std::size_t start = (*itr).first;
214  std::size_t end = results.at((*itr).second).length();
215 
216  std::string param = results.at((*itr).second);
217  p.value.insert(start, param);
218 
219  p.value.erase(start + end, 2);
220  p.value.erase(start - 1, 1);
221  }
222 
223  return p.value;
224  }
225 
226  // Log a warning.
227  return std::string();
228  }
