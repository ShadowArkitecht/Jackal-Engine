209  template <typename... Args>
210  std::string Properties::get(const std::string& name, Args&&... args) const
211  {
212  auto itr = m_properties.find(name);
213 
214  if (itr != std::end(m_properties))
215  {
216  Property_t p = itr->second;
217 
218  std::stringstream ss;
219  std::vector<std::string> results;
220 
221  this->process(ss, args...);
222 
223  while (ss.good())
224  {
225  std::string sub;
226  std::getline(ss, sub, ',');
227  results.push_back(sub);
228  }
229 
230  for (auto itr = std::rbegin(p.parameters); itr != std::rend(p.parameters); ++itr)
231  {
232  std::size_t start = (*itr).first;
233  std::size_t end = results.at((*itr).second).length();
234 
235  std::string param = results.at((*itr).second);
236  p.value.insert(start, param);
237 
238  p.value.erase(start + end, 2);
239  p.value.erase(start - 1, 1);
240  }
241 
242  return p.value;
243  }
244 
245  // Log a warning.
246  return std::string();
247  }
